{# 过滤器的使用 #}

列表反转：{{ [3,2,5,6,1] | lss }}

默认输出：{{ [3,2,5,6,1] }}

求和：{{ [3,2,5,6,1] | sum }}

排序：{{ [3,2,5,6,1] | sort}}

第一个值：{{ [3,2,5,6,1] | first }}

最后一个值：{{ [3,2,5,6,1] | last }}

长度：{{ [3,2,5,6,1] | length }}

{# 模板把变量中的特殊字符<&'">，当成原始字符串输出，不按照超文本标记语言的语义解析 #}

默认输出：{{ 'hello world' }}

{# safe是关闭模板的转义功能，谨慎使用，确认数据安全的情况下可以使用safe #}

禁用转义：{{ 'hello world' | safe }}

默认输出：{{ 'bAnK OF cHiNA' }}

大写：{{ 'bAnK OF cHiNA' | upper }}

小写：{{ 'bAnK OF cHiNA' | lower }}

首字母大写：{{ 'bAnK OF cHiNA' | capitalize }}

每个单词大写：{{ 'bAnK OF cHiNA' | title }}

字符串反转：{{ 'bAnK OF cHiNA' | reverse }}

{# 过滤器支持链式调用，执行顺序是从前到后，依次执行 #}

字符串反转再反转：{{ 'bAnK OF cHiNA' | reverse | reverse }}
